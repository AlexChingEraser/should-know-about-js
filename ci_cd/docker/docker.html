One thing confused me first is how node_modules attached in application, C# will have dot-net build/publish, C have gcc, Java have javac, but how Node.js publish with these node packages? well, some practise list here:

  1. multi-stage build
FROM node:14.4.0 AS build

COPY . .
RUN npm ci && npm run build


FROM node:slim-14.4.0

USER node
EXPOSE 8080

COPY --from=build /home/node/app/dist /home/node/app/package.json /home/node/app/package-lock.json ./
RUN npm ci --production

CMD [ "node", "dist/app.js" ]
